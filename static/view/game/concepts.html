typedef class OpConcept OpConcept;
typedef class StaticConcept StaticConcept;
typedef class PatternConcept PatternConcept;

class Concept
{
public:
    Concept();
    ~Concept();

public:
    uint16_t type; // concept*s

    tlist *netConcepts(tlist *sources); // multi activator
    DMap *net; // net(Key(concept)) = tlist

};

class OpConcept: public Concept
{
public:
    OpConcept();
    ~OpConcept();

public:
    uint16_t type; // static programmed map to classdef* opers?
    int inputs; // 0 = none, -1 = many, n = n
};

class StaticConcept: public Concept
{
public:
    StaticConcept();
    ~StaticConcept();

public:
    sint32_t type; // classdef*s
    void *data;
};

class PatternConcept: public Concept
{
public:
    PatternConcept();
    ~PatternConcept();

public:
    tlist *concepts; // sub PatternConcept*s, OpConcept*s, and StaticConcept*s
};
/* Pattern operations

{

    ~WorkspaceConcept();
